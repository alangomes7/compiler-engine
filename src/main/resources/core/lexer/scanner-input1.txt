function calculate_sum() {
    limit = 100
    current_value = 0.5
    greeting = "Starting loop\n"
    is_valid = 1
    
    if (limit >= 10) {
        limit = limit - 1
    } else {
        limit = limit + 2 * 5
    }
    
    while (limit != 0) {
        current_value = current_value / 2.0
        limit = limit - 1
    }
}
