# =========================
# Basic Math Expressions LL(1)
# =========================
expression ::= term expr_tail

expr_tail  ::= '+' term expr_tail 
             | '-' term expr_tail 
             | ε

term       ::= factor term_tail

term_tail  ::= '*' factor term_tail 
             | '/' factor term_tail 
             | ε

factor     ::= 'INT' 
             | '(' expression ')'
