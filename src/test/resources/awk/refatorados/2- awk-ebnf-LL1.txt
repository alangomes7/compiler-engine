awk-program
    ::= program-parts

program-parts
    ::= program-part program-parts
      | "newline" program-parts
      | ε

program-part
    ::= top-level-element
      | comment

comment
    ::= "#"

top-level-element
    ::= "function" "identifier" '(' opt-identifier-list ')' action
      | "BEGIN" action
      | "END" action
      | "regex-const" action
      | expression expression-construct-tail
      | action

expression-construct-tail
    ::= ',' expression action
      | action
      | terminate

opt-identifier-list
    ::= "identifier" identifier-list-tail
      | ε

identifier-list-tail
    ::= ',' "identifier" identifier-list-tail
      | ε

action
    ::= '{' statement-list '}'

statement-list
    ::= statement statement-list-tail

statement-list-tail
    ::= statement statement-list-tail
      | ε

statement
    ::= "print" opt-expr-list print-redirection terminate
      | "printf" "string-const" opt-comma-expr-list print-redirection terminate
      | "for" '(' for-content ')' action
      | "do" action "while" '(' expression ')' terminate
      | "while" '(' expression ')' action
      | "if" '(' expression ')' action else-part opt-fi
      | "switch" '(' expression ')' switch-action
      | "break" terminate
      | "continue" terminate
      | "next" terminate
      | "return" opt-return-expr terminate
      | "delete" delete-target terminate
      | expression terminate
      | action
      | terminate

delete-target
    ::= "identifier" delete-target-tail

delete-target-tail
    ::= '[' expr-list ']'
      | ε

switch-action
    ::= '{' switch-pad switch-sections '}'

switch-pad
    ::= "newline" switch-pad
      | ε

switch-sections
    ::= switch-section switch-sections
      | ε

switch-section
    ::= switch-label switch-statement-list

switch-label
    ::= "case" expression ':'
      | "default" ':'

switch-statement-list
    ::= switch-pad switch-statement-list-core

switch-statement-list-core
    ::= switch-statement switch-statement-list
      | ε

switch-statement
    ::= "print" opt-expr-list print-redirection terminate
      | "printf" "string-const" opt-comma-expr-list print-redirection terminate
      | "for" '(' for-content ')' action
      | "do" action "while" '(' expression ')' terminate
      | "while" '(' expression ')' action
      | "if" '(' expression ')' action else-part opt-fi
      | "switch" '(' expression ')' switch-action
      | "break" terminate
      | "continue" terminate
      | "next" terminate
      | "return" opt-return-expr terminate
      | "delete" delete-target terminate
      | expression terminate
      | action

terminate ::= ';' | "newline"

else-part
    ::= "else" action
      | ε

opt-fi
    ::= "fi"
      | ε

for-content
    ::= opt-expr-list for-content-tail

for-content-tail
    ::= ';' opt-expr-list ';' opt-expr-list
      | ε

opt-return-expr
    ::= expression
      | ε

opt-comma-expr-list
    ::= ',' expr-list
      | ε

print-redirection
    ::= ">>" expression
      | ε

opt-expr-list
    ::= expression expr-list-tail
      | ε

expr-list
    ::= expression expr-list-tail

expr-list-tail
    ::= ',' expression expr-list-tail
      | ε

expression
    ::= assignment expression-pipe-tail

expression-pipe-tail
    ::= '|' assignment expression-pipe-tail
      | ε

assignment
    ::= conditional assignment-tail

assignment-tail
    ::= "^=" assignment
      | "%=" assignment
      | "*=" assignment
      | "/=" assignment
      | "+=" assignment
      | "-=" assignment
      | "=" assignment
      | ε

conditional
    ::= logical-or conditional-tail

conditional-tail
    ::= '?' expression ':' conditional
      | ε

logical-or
    ::= logical-and logical-or-tail

logical-or-tail
    ::= "||" logical-and logical-or-tail
      | ε

logical-and
    ::= in-expr logical-and-tail

logical-and-tail
    ::= "&&" in-expr logical-and-tail
      | ε

in-expr
    ::= match-expr in-tail

in-tail
    ::= "in" "identifier"
      | ε

match-expr
    ::= relop-expr match-tail

match-tail
    ::= '~' match-rhs match-tail
      | "!~" match-rhs match-tail
      | ε

match-rhs
    ::= "regex-const"
      | relop-expr

relop-expr
    ::= concat-expr relop-tail

relop-tail
    ::= '<' concat-expr relop-tail
      | "<=" concat-expr relop-tail
      | "!=" concat-expr relop-tail
      | "==" concat-expr relop-tail
      | '>' concat-expr relop-tail
      | ">=" concat-expr relop-tail
      | ε

concat-expr
    ::= additive-expr concat-tail

concat-tail
    ::= concat-factor concat-tail
      | ε

concat-factor
    ::= "identifier" concat-id-tail
      | "number"
      | "string-const"
      | "regex-const"
      | "getline"
      | "DOLLAR" primary

concat-id-tail
    ::= '(' opt-expr-list ')'
      | '[' expr-list ']'
      | ε

additive-expr
    ::= multiplicative-expr additive-tail

additive-tail
    ::= '+' multiplicative-expr additive-tail
      | '-' multiplicative-expr additive-tail
      | ε

multiplicative-expr
    ::= exponential-expr multiplicative-tail

multiplicative-tail
    ::= '*' exponential-expr multiplicative-tail
      | '/' exponential-expr multiplicative-tail
      | '%' exponential-expr multiplicative-tail
      | ε

exponential-expr
    ::= unary-expr exponential-tail

exponential-tail
    ::= '^' exponential-expr
      | ε

unary-expr
    ::= '+' unary-expr
      | '-' unary-expr
      | '!' unary-expr
      | "++_pre" unary-expr
      | "--_pre" unary-expr
      | primary postfix-tail

postfix-tail
    ::= "++_post" postfix-tail
      | "--_post" postfix-tail
      | ε

primary
    ::= "identifier" primary-id-tail
      | "DOLLAR" primary
      | "number"
      | "string-const"
      | "getline"
      | '(' expression ')'

primary-id-tail
    ::= '(' opt-expr-list ')'
      | '[' expr-list ']'
      | ε
