# Tipo da linguagem: Livre de contexto (Tipo 2 – Context-Free).
# Ambiguidade: Eliminada (principalmente o dangling else).
# Recursão: Mantida apenas onde é semanticamente necessária (estruturas aninhadas e operadores). Não existe mais recursão à esquerda; toda recursão é à direita (ótima para parsers descendentes).
# A gramática corrigida gera exatamente a mesma linguagem AWK, mas agora é unambígua e mais fácil de implementar em um parser.

awk-program          ::= { top-level-element | comment }

top-level-element    ::= function-definition
                       | pattern-action-pair

function-definition  ::= "function" identifier '(' [ identifier-list ] ')' compound-statements

pattern-action-pair  ::= [ pattern ] action

pattern              ::= expression-range
                       | "BEGIN"
                       | "END"

expression-range     ::= expression [ ',' expression ]

action               ::= compound-statements

compound-statements  ::= '{' statement-list '}'

statement-list       ::= statement { terminate statement }

statement            ::= expression-statement
                       | print-statement
                       | printf-statement
                       | for-statement
                       | do-while-statement
                       | while-statement
                       | if-statement                ; agora unambiguous
                       | switch-statement
                       | controlflow-statement
                       | memory-statement

memory-statement     ::= "delete" variable

controlflow-statement::= "break"
                       | "continue"
                       | "next"
                       | "return" expression

switch-statement     ::= "switch" '(' expression ')' compound-labeled

; === IF UNAMBÍGUO (eliminada dangling-else) ===
if-statement         ::= if-without-else
                       | if-with-else

if-without-else      ::= "if" '(' expression ')' compound-statements

if-with-else         ::= "if" '(' expression ')' compound-statements
                         "else" ( if-with-else | compound-statements )

; === LOOPs ===
while-statement      ::= "while" '(' expression ')' compound-statements
do-while-statement   ::= "do" compound-statements "while" '(' expression ')'
for-statement        ::= "for" '(' [ expr-list [ ';' expr-list ';' expr-list ] ] ')' compound-statements

; === PRINT ===
print-statement      ::= "print" expr-list [ output-redirection ] terminate
printf-statement     ::= "printf" format-string ',' expr-list [ output-redirection ] terminate

expression-statement ::= expression terminate

output-redirection   ::= '>' string-const
                       | ">>" string-const
                       | '|' string-const

; === EXPRESSÕES (recursão à direita mantida, agora completa e sem ambiguidade) ===
expression           ::= binary-expr
                       | unary-expr
                       | assign-expr
                       | ternary-expr

assign-expr          ::= variable ( "^=" | "%=" | "*=" | "/=" | "+=" | "-=" | "=" ) expression

ternary-expr         ::= binary-expr '?' binary-expr ':' binary-expr

binary-expr          ::= logical-or-expr

logical-or-expr      ::= logical-and-expr ( "||" logical-or-expr )?          ; permite múltiplos ||
logical-and-expr     ::= in-expr ( "&&" logical-and-expr )?                 ; recursão à direita
in-expr              ::= match-expr ( "in" variable )?
match-expr           ::= relop-expr ( ( '~' | "!~" ) match-expr )?          ; recursão à direita
relop-expr           ::= concat-expr ( ( '<' | "<=" | "!=" | "==" | '>' | ">=" ) relop-expr )?
concat-expr          ::= additive-expr concat-expr?                          ; concatenação implícita
additive-expr        ::= multiplicative-expr ( ( '+' | '-' ) additive-expr )?
multiplicative-expr  ::= exponential-expr ( ( '*' | '/' | '%' ) multiplicative-expr )?
exponential-expr     ::= unary-expr ( '^' exponential-expr )?                ; ^ é associativo à direita

unary-expr           ::= ( '+' | '-' | '!' | '~' | '$' | "++" | "--" ) primary
                       | postfix-expr

postfix-expr         ::= primary ( "++" | "--" )

primary              ::= variable
                       | const-literal
                       | getline
                       | identifier '(' [ expr-list ] ')'
                       | '(' expression ')'

getline              ::= [ string-const '|' ] "getline" [ variable ] [ '<' string-const ]

; === TIPOS BÁSICOS (sem alteração) ===
variable             ::= identifier [ index ]
index                ::= '[' { alphanumeric } ']'
identifier-list      ::= identifier { ',' identifier }
identifier           ::= alphabetic { alphanumeric | '_' }

; ... (const-literal, regex-const, string-const, números, format-string, comment, terminate, etc. permanecem iguais à original)

const-literal		::= regex-const
			| string-const
			| float-const
			| integer-const

regex-const		::= '/' { vis-glyph } '/'

string-const		::= '"' { vis-glyph } '"'
			| "'" { vis-glyph } "'"

float-const		::= [ dec-integer ] '.' dec-integer

integer-const		::= dec-integer
			| hex-integer
			| oct-integer
			| bin-integer

dec-integer		::= int-digit { int-digit }

hex-integer		::= ( "0x" | "0X" ) hex-digit { hex-digit }

oct-integer		::= ( "0o" | "0O" ) oct-digit { oct-digit }

bin-integer		::= ( "0b" | "0B" ) bin-digit { bin-digit }

hex-digit		::= int-digit | 'a' | 'b' | 'c' | 'd' | 'e' | 'f'
				     | 'A' | 'B' | 'C' | 'D' | 'E' | 'F'

int-digit		::= oct-digit | '8' | '9'

oct-digit		::= bin-digit | '2' | '3' | '4' | '5' | '6' | '7'

bin-digit		::= '0' | '1' 

format-string		::= '"' { format-glyph } '"'
			| "'" { format-glyph } "'"

comment			::= "# " { no-newline } sys-newline

terminate		::= [ ';' ] sys-newline
