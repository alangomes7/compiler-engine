# =========================
# Program (multiple lines)
# =========================
program        ::= line program
                 | ε

line           ::= signed_number NEWLINE

# =========================
# Signed Number
# =========================
signed_number  ::= sign_opt number

sign_opt       ::= '+' 
                 | '-' 
                 | ε

number         ::= 'SCIENTIFIC'
                 | 'FLOAT'
                 | 'INT'
